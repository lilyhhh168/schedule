// EduSchedule GAS v4 const SS=SpreadsheetApp.getActiveSpreadsheet(); function doGet(e){return handle(e);} function doPost(e){return handle(e);} function handle(e){ const p=e.parameter; let r={}; try{ if(p.action==='read'){ r={teachers:rd('teachers'),students:rd('students'), classes:rd('classes'),bonuses:rd('bonuses')}; }else if(p.action==='write'){ const d=JSON.parse(p.payload); Object.keys(d).forEach(k=>ws(k,d[k])); r={ok:true}; } }catch(ex){r={error:ex.toString()};} return ContentService.createTextOutput(JSON.stringify(r)) .setMimeType(ContentService.MimeType.JSON); } function rd(name){ const s=SS.getSheetByName(name); if(!s)return[]; const rows=s.getDataRange().getValues(); if(rows.length<2)return[]; const h=rows[0]; return rows.slice(1).map(r=>{ const o={}; h.forEach((k,i)=>{ let v=r[i]; if(typeof v==='string'&&(v.startsWith('[')||v.startsWith('{'))){ try{v=JSON.parse(v);}catch(e){} } o[k]=v; }); return o; }); } function ws(name,data){ let s=SS.getSheetByName(name); if(!s)s=SS.insertSheet(name); s.clearContents(); if(!data||!data.length)return; const h=Object.keys(data[0]); s.getRange(1,1,1,h.length).setValues([h]); s.getRange(2,1,data.length,h.length).setValues( data.map(o=>h.map(k=>{ const v=o[k]; return typeof v==='object'&&v!==null?JSON.stringify(v):(v??''); })) ); }
